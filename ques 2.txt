#include<bits/stdc++.h>
using namespace std;
void cou(int ar[20],int money)
{
    int count=0;
    if(money==2000){
    cout<<"1";
    }
    while(money>0){
        for(int i=0;i<10;i++)
        {
            if(ar[i]>money){
                money=money-ar[i-1];
                count++;
                break;
            }
        }
    }
   cout<<count;
}
int main()
{
    int ar[]={1,2,5,10,20,50,200,500,2000};
    int money;
    cin>>money;
    cou(ar,money);


}
